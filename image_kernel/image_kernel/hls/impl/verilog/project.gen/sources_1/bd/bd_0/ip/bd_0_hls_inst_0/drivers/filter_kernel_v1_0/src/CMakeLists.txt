cmake_minimum_required(VERSION 3.15)
project(filter_kernel)
find_package(common)
collector_create (PROJECT_LIB_SOURCES "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_SOURCES xfilter_kernel.c)
collect (PROJECT_LIB_SOURCES xfilter_kernel_sinit.c)
collect (PROJECT_LIB_SOURCES xfilter_kernel_linux.c)
collect (PROJECT_LIB_SOURCES xfilter_kernel_g.c)
collector_list (_sources PROJECT_LIB_SOURCES)
collector_create (PROJECT_LIB_HEADERS "${CMAKE_CURRENT_SOURCE_DIR}")
collect (PROJECT_LIB_HEADERS xfilter_kernel.h)
collect (PROJECT_LIB_HEADERS xfilter_kernel_hw.h)
collector_list (_headers PROJECT_LIB_HEADERS)
file(COPY ${_headers} DESTINATION ${CMAKE_BINARY_DIR}/include)
include_directories(${CMAKE_BINARY_DIR}/include)
add_library(filter_kernel STATIC ${_sources})
set_target_properties(filter_kernel PROPERTIES LINKER_LANGUAGE C)
